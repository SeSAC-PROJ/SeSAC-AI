# 1. 베이스 이미지
FROM python:3.10

# 2. 시스템 의존성 설치 (ffmpeg, rust, build-essential 등)
RUN apt-get update && apt-get install -y --no-install-recommends ffmpeg curl build-essential libopenblas-dev && rm -rf /var/lib/apt/lists/*
RUN curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh -s -- -y
ENV PATH="/root/.cargo/bin:${PATH}"

# 3. 작업 디렉토리 설정
WORKDIR /app

# 4. Poetry 설치
RUN pip install --upgrade pip && pip install poetry

# 5. 의존성 정의 파일 복사
COPY pyproject.toml poetry.lock* ./

# 6. Poetry를 사용한 프로젝트 의존성 설치
# --no-root: 프로젝트 자체는 패키지로 설치 안 함
# --no-dev: 개발용 의존성은 건너뜀
RUN poetry install --no-root --no-dev

# 7. 애플리케이션 코드 복사
COPY ./app ./app

# 8. 애플리케이션 실행 명령어
CMD ["poetry", "run", "uvicorn", "app.main:app", "--host", "0.0.0.0", "--port", "8000"]
